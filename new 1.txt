import webbrowser
import time

# URL you want to refresh
url = 'https://invite-c4ct.onrender.com/'  # replace with your webpage

# How many seconds between refreshes
refresh_interval = 10  # every 10 seconds

# How many times to refresh
refresh_count = 5

for i in range(refresh_count):
    webbrowser.open(url, new=0)  # open in same browser window if possible
    print(f"Refreshed {i + 1} time(s)")
    time.sleep(refresh_interval)
